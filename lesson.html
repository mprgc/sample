11 Sum and Average Example
PROGRAM sum2;
VAR x, total, count: integer; average: real;
BEGIN
  total := 0; count := 0;
  write('Enter number: ');
  read(x);
  WHILE (x < 10) DO
  BEGIN
    count := count + 1;
    total := total + x;
    write('Enter number: ');
    read(x);
  END;
  average := total / count;
  writeln;
  writeln('Total = ', total);
  writeln('Average = ', average:0:1);
END.
